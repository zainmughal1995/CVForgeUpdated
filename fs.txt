main.jsx
 └── App.jsx
      └── AppRouter.jsx
           ├── /login → Login.jsx
           │        └── useAuth()
           │              └── authService.login()
           │                    └── dispatch(setUser)
           │
           ├── /register → Register.jsx
           │        └── useAuth()
           │              └── authService.register()
           │
           └── /builder → ProtectedRoute
                    ├── checks authSlice (isAuthenticated)
                    └── if true → Builder.jsx
                              └── BuilderLayout.jsx
                                   ├── Navbar.jsx
                                   │      └── useAuth()
                                   │            └── logout()
                                   │
                                   ├── SectionsSidebar.jsx
                                   │      └── sets active section (local state)
                                   │
                                   ├── Dynamic Form (based on section)
                                   │      └── useCV()
                                   │            └── dispatch(update/add/remove)
                                   │
                                   └── CVPreview.jsx
                                          └── useSelector(cv state)
